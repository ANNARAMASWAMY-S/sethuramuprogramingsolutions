DECLARE
    CURSOR cur_vip IS
        SELECT CustomerID
        FROM Customers
        WHERE Balance > 10000;

BEGIN
    FOR rec IN cur_vip LOOP
        UPDATE Customers
        SET IsVIP = 'Y'
        WHERE CustomerID = rec.CustomerID;

        DBMS_OUTPUT.PUT_LINE('VIP status updated for Customer ID: ' || rec.CustomerID);
    END LOOP;
END;
/
